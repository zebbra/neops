    Cron job browser.service

    For future reference: Could adapt the service to our service structure for entities, something like:

    import { Injectable } from '@angular/core'
import { NeopsBaseSearchService } from './element-search/service/base/neops-base-search-service'

import {
  NeopsTaskCronjobsGQL,
  NeopsTaskCronjobsQuery,
  NeopsTaskCronjobsQueryVariables,
  NeopsTaskCronJobType,
  ScopeType,
} from '@zebbra/neops-angular-graphql-client'
import { AlertService } from '@zebbra/ngx-neops-app-services'
import { ContextService } from './common/context.service'
import { OrderConfig } from './common/order-config'
import { NeopsTaskCronJob } from './model/NeopsTaskCronJob'
import { ElasticSearchOnJsonDataService } from './elastic-search-on-json-data.service'
import { firstValueFrom } from 'rxjs'

const DEFAULT_PAGE_LENGTH = 10
const DEFAULT_PAGE = 1

/**
 *
 */
@Injectable({ providedIn: 'root' })
export class CronJobBrowserService extends NeopsBaseSearchService<
  NeopsTaskCronJob,
  NeopsTaskCronjobsQuery,
  NeopsTaskCronjobsQueryVariables
> {
  private orderConfig?: OrderConfig
  private pageNumber: number = DEFAULT_PAGE
  private pageSize: number = DEFAULT_PAGE_LENGTH
  private query: string = '*'
  /**
   * Default constructor
   *
   * @param cronJobsGQL
   * @param alertService
   * @param contextService
   */
  constructor(cronJobsGQL: NeopsTaskCronjobsGQL, alertService: AlertService, contextService: ContextService) {
    super(cronJobsGQL, alertService, contextService)
  }

  protected async extractResults(searchResult: NeopsTaskCronjobsQuery): Promise<NeopsTaskCronJob[] | null | undefined> {
    const start = (this.pageNumber - 1) * this.pageSize
    const res = searchResult.neopsTaskCronjob?.results?.map(res => new NeopsTaskCronJob(res as NeopsTaskCronJobType))

    const s = new ElasticSearchOnJsonDataService(res)
    s.applyOrderConfig(this.orderConfig || {})

    const sorted = s.searchElements()

    const elements = await firstValueFrom(sorted.elements)

    const paginated = elements?.slice(start, start + this.pageSize)

    return new Promise(r => paginated)
  }

  protected extractTotalCount(searchResult: NeopsTaskCronjobsQuery): number | null | undefined {
    return searchResult.neopsTaskCronjob?.results?.length || 0
  }

  protected getDefaultSearchVariables(): NeopsTaskCronjobsQueryVariables {
    return {}
  }

  protected setOrdering(partialQuery: {}, orderConfig: OrderConfig): {} {
    this.orderConfig = orderConfig
    return partialQuery
  }

  protected setPage(
    partialQuery: Partial<NeopsTaskCronjobsQueryVariables>,
    pageNumber: number
  ): Partial<NeopsTaskCronjobsQueryVariables> {
    this.pageNumber = pageNumber
    return partialQuery
  }

  protected setPageSize(
    partialQuery: Partial<NeopsTaskCronjobsQueryVariables>,
    pageSize: number
  ): Partial<NeopsTaskCronjobsQueryVariables> {
    this.pageSize = pageSize
    return partialQuery
  }

  protected setQuery(
    partialQuery: Partial<NeopsTaskCronjobsQueryVariables>,
    query: string
  ): Partial<NeopsTaskCronjobsQueryVariables> {
    this.query = query
    return partialQuery
  }

  protected setScope(
    partialQuery: Partial<NeopsTaskCronjobsQueryVariables>,
    scope: ScopeType
  ): Partial<NeopsTaskCronjobsQueryVariables> {
    return partialQuery
  }
}
